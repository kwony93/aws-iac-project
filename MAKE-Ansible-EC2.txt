### AWS에 EC2 생성
생성 후 VSCode를 이용한 코드 작업 시 t3.medium으로 생성할 것   
t3.micro는 메모리 Memory 부족으로 접속 잘 안됨


# Ansible 설치

// 시스템 업데이트   
sudo dnf update -y   
sudo dnf install python3-pip -y   

// pip 업데이트   
python3 -m pip install --upgrade pip   

// anisble 설치   
python3 -m pip install ansible   

// packaging 모듈 설치   
python3 -m pip install packaging   

// boto3와 botocore 설치   
python3 -m pip install boto3 botocore   

// PATH에 ~/.local/bin 추가   
echo 'export PATH=$PATH:$HOME/.local/bin' >> ~/.bashrc   

// 변경사항 즉시 적용   
source ~/.bashrc   

// 확인   
ansible-playbook --version   

// ansible-galaxy로 AWS 컨렉션 설치   
ansible-galaxy collection install amazon.aws community.aws   

# Project 준비   

mkdir -p project/{roles,group_vars,scripts,keys}   
touch ansible.cfg inventory.yml   

// roles 디렉터리로 이동   
cd roles   

// 각 서비스별 Role 초기화   
ansible-galaxy role init network   
ansible-galaxy role init jenkins   
ansible-galaxy role init loadbalancer   
ansible-galaxy role init asg   
ansible-galaxy role init security 

// 다시 프로젝트 루트로 복귀   
cd ..   


cd ~/.ssh
Ansible용 키 페어 생성 후 등록
vim <키페어 이름>.pem
chmod 400 <키페어 이름>.pem
